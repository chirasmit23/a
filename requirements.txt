# Requirements for UltraChat Voice RAG Application

# --- Core Application Framework ---
streamlit>=1.33.0               # The main framework for building the web app UI.
python-dotenv>=1.0.0            # For loading secret API keys from a .env file.

# --- AI & Language Models ---

# Note: This will automatically install the necessary 'langchain' and 'langchain-core' packages.

# --- Voice & Audio Processing ---

# --- Web Scraping & API Requests ---

beautifulsoup4>=4.12.3          # For parsing HTML to extract the YouTube transcript.
lxml>=5.2.2                     # A fast and efficient parser for BeautifulSoup.

# --- Document & File Processing ---
# For handling uploaded files like .docx, images, and .pdf.
python-docx>=1.1.2              # Required by docx2txt to handle Microsoft Word files.
docx2txt>=0.8                   # For extracting text from .docx files.
Pillow>=10.0.0                  # For opening and processing image files.
pytesseract>=0.3.10             # For performing OCR (Optical Character Recognition) to extract text from images.
PyMuPDF>=1.24.0                 # A robust library for extracting text and data from .pdf files.
